\begin{frame}
	\frametitle{Consistency}
	% Two Types: Raft for Consensus, Serf for Gossip
	\begin{columns}[b]
		\begin{column}{0.5\textwidth}
			\emph{Raft} Protocol (Consensus)
			\begin{itemize}
				\item Replicated Cluster State
				\item Example: KV store, service and node IP addresses, configuration
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\emph{Surf} Protocol (Gossip)
			\begin{itemize}
				\item Distribute and perform \emph{node health checks}
				\item Examples for health checks: HTTP GET Request, gRPC, TCP, UDP
				% https://developer.hashicorp.com/consul/docs/services/usage/checks
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

		\item Three states
	\end{itemize}

\begin{frame}
		\frametitle{Consistency}

		\begin{itemize}
			\item \emph{Writes}: always sent to leader
			\item \emph{Reads}: three consistency modes:
			\begin{description}[labelsep=0.5em]
				\setlength\itemsep{0.5em}
				\item[default] \emph{leader leasing}: leader assumes its role for a specific time window and responds without quorum
				\\
				(However: risk of 2 concurrent leaders $\Rightarrow$ stale reads)
				\item[consistent] leader has to verify its role before responding
%				that they are still the leader 
				\item[stale] any server agent (leader \& follower) can respond
			\end{description}
		\end{itemize}

				%\textcite[Consensus Protocol]{ConsulDoc}
\end{frame}

\subsection{Usage}

\begin{frame}[fragile]
	\frametitle{CLI agent}

	\begin{columns}[t]
		\begin{column}{0.45\textwidth}
			\begin{minted}{bash}
$ consul agent -dev

# other terminal:
$ consul kv put my/key 123
$ consul kv get my/key
123
			\end{minted}
		\end{column}
		\begin{column}{0.55\textwidth}
			CLI or HTTP API

			Responses contain metadata like last writer, ...
			% Metadata response: https://developer.hashicorp.com/consul/api-docs/kv#metadata-response
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Long Polling}
	\begin{minted}{bash}
$ curl -v http://localhost:8500/v1/kv/my/key

< X-Consul-Index: 19
# ...

$ curl -v http://localhost:8500/v1/kv/my/key?index=19
# blocks until value is changed or timeout is reached
# (max. 10 minutes)
	\end{minted}
\end{frame}
